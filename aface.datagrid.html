        // 默认值
        defaultOptions : {
            selectedTabIndex : 0,
            onSelectedTabChange : null
        }
        
        // 初始化
        $('#tabs').tabs({
            // 初始化时默认选中Tab的索引
            selectedTabIndex : 0,
            /**
                * 选中Tab发生变化时触发的事件
                * 
                * @param index: 选中的Tab的索引
                * @param event: 鼠标单击事件
                */
            onSelectedTabChange : function(index, event) {
                console.log('selectedTabIndex: ' + index);
            }
        });
        
        // 获取指定dom对象上绑定的js对象
        var tabs = $('#tabs').target('tabs');
        
        /**
            * 获取当前选中的Tab的索引
            * 
            * @return: 0 索引
            */
        tabs.getSelectedTabIndex();
        
        /**
            * 按索引选中Tab
            * 
            * @param idx: 0 要选中的Tab的索引
            */
        tabs.selectByIndex(1);
